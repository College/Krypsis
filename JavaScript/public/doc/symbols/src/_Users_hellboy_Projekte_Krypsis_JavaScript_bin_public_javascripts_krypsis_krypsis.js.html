 37   /**
 38    * <p>
 39    * Load the Krypsis framework. This function must be added to the body.unload event.
 40    * </p>
 41    *
 42    * <p>
 43    * Due to some mobile-browser limitations Krypsis is not able to register
 44    * itself for a "DOM is ready" event. Therefore you, as a web developer, have
 45    * to do it by yourself. Best practice is to wire the onload and onunload events
 46    * of the body element with the Krypsis.load and Krypsis.unload functions.
 47    * </p>
 48    *
 49    * <p>
 50    * If you need to get notified about those events you can add an onLoad and
 51    * onUnload listener.
 52    * @link Krypsis#addLoadEvent
 53    * @link Krypsis#addUnloadEvent
 54    * </p>
 55    *
 56    * @example
 57    * <pre>
 58    *  <body onload="Krypsis.load();"/>
 59    * </pre>
 60    * 
 61    * @public
 62    */
 63   load: function() {
 64     this.loaded = true;
 65     this.fireEvent(this, 'onLoad');
 66 
 67     if (!Krypsis.isSet(Krypsis.Device.connector)) {
 68       Krypsis.Device.connector = Krypsis.Device.guessConnector();
 69       Krypsis.Log.info("Set device connector to " + Krypsis.Device.connector);
 70     }
 71 
 72     Krypsis.Log.info("Krypsis is successfully loaded");
 73   },

 76   /**
 77    * <p>
 78    * Unload Krypsis. This function must be added to the body.onunload event.
 79    * </p>
 80    *
 81    * <p>
 82    * The expected parameter tells Krypsis whether the underlying device should
 83    * be notified or not. If set to true, then the Krypsis.Device.Base.reset();
 84    * function will be executed and the underlying device will stop and
 85    * collect all previously created listeners. Also the device internal state
 86    * will be reseted. This is very recommended, since the user can cancel using
 87    * the web application and all created listeners will be active in backgroud,
 88    * which can result in unmeant behaviuor.
 89    * </p>
 90    *
 91    * <p>
 92    * So don't forget to provide the unload function in the body-tag callback
 93    * </p>
 94    *
 95    * @example
 96    * <pre>
 97    *  <body onunload="Krypsis.unload(true);"/>
 98    * </pre>
 99    *
100    * @param {Function} notifyDevice Should the device be notified? Default is set to TRUE.
101    * @public
102    */
103   unload: function(notifyDevice) {
104     this.loaded = false;
105 
106     if (Krypsis.isSet(notifyDevice) ? notifyDevice : true) {
107       Krypsis.Device.Base.reset({});
108     }
109 
110     this.fireEvent(this, 'unUnload');
111 
112     Krypsis.Log.info("Krypsis is successfully unloaded");
113   },

153   /**
154    * <p>
155    * Adds a listener to an object to observe the execution of a particular
156    * function.
157    * </p>
158    *
159    * @param {Function} callback The function to call when the observed method is executed.
160    * @param {Object} obj The object that provides the observed method
161    * @param {String} name The name of the method to observe
162    *
163    * @public
164    */
165   addListener: function(callback, obj, name) {
166     if (typeof callback != 'function') {
167       throw new Error("Function is expected as parameter");
168     }
169 
170     var oldListener = obj[name] || function() {};
171     obj[name] = function() {
172       var args = Array.prototype.slice.call(arguments);
173 
174       oldListener.apply(obj, args);
175       callback.apply(obj, args);
176     };
177   },

179   /**
180    * <p>
181    * Executes a function of the given object. This function is used
182    * whenever a listener was previosly added and now the source event
183    * should be fired.
184    * </p>
185    *
186    * <p>
187    * For example: The load event if Krypsis is defined as follows
188    * <code>Krypsis.addListener(FUNCTION, this, 'onLoad');</code>
189    * If now all listener should be notified about this 'load' event
190    * then we can achieve this by execution this function:
191    * <code>Krypsis.fireEvent(Krypsis, 'onLoad');</code>
192    * </p>
193    *
194    * <p>
195    * Alternatively we could write:
196    * <pre>
197    * if (Krypsis.isSet(Krypsis.onLoad)) {
198    *  Krypsis.onLoad();
199    * }
200    * </pre>
201    * </p>
202    *
203    * @param {Object} source The source object
204    * @param {String} event The function to execute as string.
205    *
206    * @private
207    */
208   fireEvent: function(source, event) {
209     if (!source || ! event) {
210       return false;
211     }
212 
213     var eventFunction = source[event];
214     if (typeof eventFunction == 'function') {
215       var args = Array.prototype.slice.call(arguments);
216       if (args.length >= 2) {
217         args = args.slice(2, args.length);
218       }
219       eventFunction.apply(source, args);
220     }
221   },

224   /**
225    * <p>
226    * Returns the element with the given id.
227    * </p>
228    * 
229    * @param {String|Object} id The id of the element
230    * @param {Boolean} raise If it set to true and the element
231    * was not found then an exception will be thrown
232    *
233    * @public

235   $: function(id, raise) {
236 
237     if ("object" == typeof id) { return id; }

239     raise = Krypsis.isSet(raise) ? raise : true;
240 
241     var element = null;
242 
243     if (Krypsis.isDefined(typeof document.getElementById)) {
244       element = document.getElementById(id);
245     }
246     else if (Krypsis.isDefined(typeof document.all)) {
247       element = document.all[id];
248     }
249     else {
250       var errorMessage = "Krypsis could not access DOM elements. The getElementById() function is not defined.";
251       Krypsis.Log.error(errorMessage);
252       throw new Error(errorMessage);
253     }
254 
255     if (!Krypsis.isSet(element) && raise) {
256       throw "Element with id " + id + " does not exist.";
257     }
258     return element;
259   },

536   /**
537    * This is a helper module to build uris
538    * parameters from hashes
539    * 
540    * @static
541    * @class The
542    */
543   Uri: {
544     /**
545      * Expects a hash of parameters that should be converted
546      * in URI compatible string and joined with a '&'.
547      * 
548      * @param {Object} parameters The parameters hash {}
549      *
550      * @public
551      */
552     toQueryString: function(parameters) {
553       if (!Krypsis.isSet(parameters) || parameters.length <= 0) {
554         return null;
555       }
556 
557       var result = this.toQueryPairs(parameters);
558       return result && result.length > 0 ? result.join('&') : null;
559     },
560 
561     /**
562      * Iterates over the given hash and creates an
563      * array of key=value pairs. The array can be joined
564      * after this procedure.
565      * Note: If the hash is undefined, null or empty
566      * the result is null.
567      *
568      * @param {Object} hash The hash to convert into key=value pairs.
569      *
570      * @private
571      */
572     toQueryPairs: function(hash) {
573       if (!hash) { return null; }
574       
575       var result = [];
576 
577       for (var key in hash) {
578         if (typeof hash[key] == 'function') {
579           continue;
580         }
581 
582         var pair = this.toQueryPair(key, hash[key]);
583         if (pair) { result.push(pair); }
584       }
585 
586       return result.length == 0 ? null : result;
587     },
588 
589     /**
590      * Encodes the given key and value to a URI compatible
591      * string and concatinates them with a '='
592      *
593      * @param {String} key The key
594      * @param {String} value The value
595      * @return {String} encoded 'key=value'
596      *
597      * @private
598      */
599     toQueryPair: function(key, value) {
600       if (!key || key.length == 0) { return null; }
601       value = Krypsis.isSet(value) ? value : '';
602       return encodeURIComponent(key) + '=' + encodeURIComponent(value);
603     }
604   },

606   /**
607    * This is a small AJAX library that enables
608    * the framework to call remote services in background
609    * without reloading the full page content
610    * 
611    * @class Krypsis AJAX library
612    * @static
613    */
614   Xhr: {
615 
616     /**
617      * Executes a request to the specified url.
618      * You can control the request process with the
619      * given options.
620      * 
621      * @param {String} url The url to call.
622      * @param {Object} options The options hash.
623      *
624      * <p>
625      * The following parameters are supported and used to control the
626      * behaviour of the AJAX transmission.
627      * <ul>
628      * <li><strong>method</strong> Either <code>POST</code> or <code>GET</code>. Default is <code>GET</code></li>
629      * <li><strong>parameters</strong> The parameters hash with key - value pairs to send to the server.</li>
630      * <li><strong>evalScripts</strong> Should the javascript be evalueted if any found. Default is false.</li>
631      * <li><strong>asynchronous</strong> The type of the connection to the server. If <code>false</code>, then the operation
632      * block until the server responds to the request. Default is <code>true</code></li>
633      * </ul>
634      * </p>
635      *
636      * <p>
637      * A main part of the options hash are the definable callbacks
638      * (very similar to the prototype's callbacks)
639      * <ul>
640      * <li><strong>onCreate</strong> The XHR instance is created. (1 Param => the XHR wrapper)</li>
641      * <li><strong>onInitialized</strong> Connection is opened. (1 Param => the XHR wrapper)</li>
642      * <li><strong>onSent</strong> Data is sent. (1 Param => the XHR wrapper)</li>
643      * <li><strong>onLoading</strong> Read the Response. (1 Param => the XHR wrapper)</li>
644      * <li><strong>onSuccess</strong> Response status is ok. Something between 200 and 299. (1 Param => the XHR wrapper)</li>
645      * <li><strong>onFailure</strong> Response statis is NOT ok. A status code that is not between 200 and 299. (1 Param => the XHR wrapper)</li>
646      * <li><strong>onException</strong> An exception occurs during the request. (1 param => the XHR wrapper, 2 param => Exception itself)</li>
647      * </ul>
648      * </p>
649      *
650      * @return The XHR wrapper. To access the XMLHttpRequest instance, call <code>xhr.transport</code>
651      *
652      * @public
653      */
654     request: function(url, options) {
655 
656       if (!Krypsis.isSet(url)) {
657         throw "Url is missing";
658       }
659 
660       /*
661        * Attention: this is a wrapper around the xmlhttprequest instance.
662        * Access the instance by the transport property
663        */
664       var xhr = this.getXmlHttpRequest();
665 
666       if (!Krypsis.isSet(xhr)) {
667         throw "XmlHttpRequest is not supported by this browser";
668       }
669 
670       /*
671        * Make sure the given options are valid and
672        * the default values are set.
673        */
674       options.applyDefaults({
675         method: 'GET',
676         parameters: null,
677         username: null,
678         password: null,
679         asynchronous: true,
680         evalScripts: false
681       });
682 
683       /*
684        * Apply the user callbacks by copying each function
685        * that begins with 'on' to the callbacks hash
686        */
687       this.applyCallbacks(xhr, options);
688 
689       /*
690        * Dispatches the event with the given name.
691        * If callback is not defined, nothing happens.
692        */
693       this.applyDispatchEventFunction(xhr);
694 
695       /*
696        * Dispatches the fist event => onCreate.
697        * The usecase on this callback could be to
698        * show a spinner or deactivate some elements.
699        */
700       xhr.dispatchStateEvent("Create");
701 
702       /* Prepare options */
703       var parameters = options.parameters;
704       var method = options.method;
705       xhr.evalScripts = options.evalScripts;
706 
707       this.applyOnReadyStateChangedFunction(xhr);
708 
709 
710       try {
711         /* Open connection */
712         xhr.transport.open(method, url, options.asynchronous);
713 
714         /*
715          * Compile the url and the request headers.
716          * If parameters are given then the url build depends on
717          * the method.
718          *
719          * In case of an get, we have to append the parameters to the
720          * url and send the request.
721          *
722          * In case of a post the content type has to be set to
723          * 'application/x-www-form-urlencoded' and the parameters will
724          * be sent by the underlying XMLHttpRequest instance as body content.
725          */
726         if (parameters) {
727           /* Convert the params to an url encoded string. */
728           parameters = Krypsis.Uri.toQueryString(parameters);
729           if (method == "GET") {
730             url = url = "?" + parameters;
731             parameters = null;
732           }
733           else {
734             xhr.transport.setRequestHeader("Connection", "close");
735             xhr.transport.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
736             xhr.transport.setRequestHeader("Content-length", parameters.length);
737           }
738         }
739 
740         xhr.transport.send(parameters);
741         return xhr;
742       }
743       catch (e) {
744         /**
745          * An exception we can handle in two different
746          * ways. If an onException callback is provided
747          * then the event will be dispatched.
748          *
749          * Otherwise we will rethrow the exception.
750          */
751         if (!xhr.dispatchStateEvent("Exception", e)) {
752           throw e;
753         }
754       }
755     },
756 
757     /**
758      * <p>
759      * Calls an ajax updater and replaces the given
760      * container with the result in successfull
761      * </p>
762      *
763      * @param {String|Object} container The target container where to place the received content into.
764      * The container can be a string or a hash. If the param is a string, then it will be
765      * converted into a hash where the string bacames the value of the the success key.
766      * The hash contains the folowing keys:
767      * <ul>
768      * <li><strong>success</strong> The container id in case of a success response</li>
769      * <li><strong>failure</strong> The container id in case of a failed response</li>
770      * </ul>
771      *
772      * @param {String} url The url to execute.
773      * @param {Object} options The options hash. See the {@link Krypsis.Xhr.request()} function for more
774      * infromation about the options hash
775      *
776      * @return {Object} the XHR wrapper instance
777      *
778      * @public
779      */
780     update: function(container, url, options) {
781       if (!container) {
782         throw "container is missing. Could not update a null element";
783       }
784 
785       if ("string" == typeof container) {
786         container = { success: container };
787       }
788 
789       container.success = Krypsis.$(container.success);
790       if (container.failure) {
791         container.failure = Krypsis.$(container.failure);
792       }
793 
794       options = options || {};
795 
796       var oldOnSuccess = options.onSuccess;

1635

1649 
1650             onCreate: function() {

1661        * Returns the hostname of the current loaded window instance.

1663        * suffixed with the port. The host name does not contains a trailing
1664        * slash.
1665        *
1666        * @example
1667        * http://localhost:3000
1668        *
1669        * @private

1671       getHostName: function() {
1672         if (!this.host) {
1673           /* Contains also the port */
1674           var host = window.location.host;

1685       /**
1686        * The queue is an implementation of the event driven AJAX
1687        * pattern. It opens up a connection to the krypsis implementation
1688        * and the java program holds this connection until a timeout
1689        * occurs or until a response is ready to be dispatched to the
1690        * browser.
1691        *
1692        * Each time the queue is returning back it will reconnect itself.
1693        */
1694       Queue: {
1695         /** Indicates if the queue is connectet o not*/
1696         running: false,
1697         aborted: false,
1698         queue: null,
1699 
1700         /**
1701          * Starts the queue.
1702          */
1703         start: function() {
1704           var self = this;
1705           if (self.running) { return true; }
1706           Krypsis.Log.debug("Start blackberry queue");
1707 
1708           var url = Krypsis.Device.BlackBerryConnector.namespace + "queue/start";
1709           self.queue = Krypsis.Xhr.request(url, {
1710             onSuccess: function(xhr) {
1711               try {
1712                 var text = xhr && xhr.transport ? xhr.transport.responseText : null;
1713                 if (text && text.length > 0) {
1714                   Krypsis.Log.debug("Evaluate response: " + text);
1715                   eval(text);
1716                 }
1717               }
1718               catch(e) {
1719                 Krypsis.Log.error("Could not evaluate response: " + e);
1720               }
1721             },
1722 
1723             onCreate:   function() {
1724               self.running = true;
1725               self.aborted = false; 
1726             },
1727             
1728             onComplete: function() {
1729               self.running = false;
1730               if (!self.aborted) {
1731                 self.start();
1732               }
1733             }
1734           });
1735         },
1736 
1737         /**
1738          * Stops the queue by aborting the XMLHttpRequest
1739          * instance (if exists) and setting the running flag to
1740          * false.
1741          */
1742         stop: function() {
1743           if (this.queue) {
1744             Krypsis.Log.debug("Stopping BlackBerry response queue...");
1745             this.aborted = true;
1746             this.queue.transport.abort();
1747             this.running = false;
1748             this.queue = null;
1749             Krypsis.Log.debug("Stopped");
1750           }
1751         }
1752       }
1753     }
1754   }

1782 /**
1783  * <p>
1784  * To handle the krypsis tasks in an easy way, we have extended the Array
1785  * class and added this execute() function. It expects the elements to be
1786  * instances of the Krypsis.Device.Task class and calls the execute() function
1787  * on each task. When all tasks completed their work, then the given callback
1788  * will be called.
1789  * </p>
1790  *
1791  * <p>
1792  * As you sometimes need to combine some Krypsis results,
1793  * you can use this extension as a notification container.
1794  * </p>
1795  *
1796  * @example
1797  * <pre>
1798  * var metaData = null;
1799  * var response = null;
1800  *
1801  * var tasks = [
1802  *    Krypsis.Device.Base.getMetaData({
1803  *      onSuccess: function(args) { metaData = args; }
1804  *    }),
1805  *    Krypsis.Device.Photo.takeAndUpload({
1806  *      parameters = {...},
1807  *      onSuccess: function(args) { response = args.response; }
1808  *    });
1809  * ];
1810  *
1811  * tasks.execute(function() {
1812  *    if (!metaData || !response) {
1813  *      // This never happen.
1814  *      throw new Error("Not all tasks executed");
1815  *    }
1816  * });
1817  * </pre>
1818  *
1819  * @param {Function} callback The callback to execute when all tasks are finished.
1820  */
1821 Array.prototype.execute = function(callback) {
1822   var self = this;
1823   self.tasksCompleted = false;
1824 
1825   if (callback) {
1826     Krypsis.addListener(callback, self, 'onTasksCompletedEvent');
1827   }
1828 
1829   /**
1830    * Will be called, when a task is finished.
1831    * Internaly a counter will be decreased and when it
1832    * reaches zero, then this execution is completed.
1833    *
1834    * @private
1835    */
1836   self.taskCompleted = function () {
1837 
1838     if (self.tasksCompleted) {
1839       return false;
1840     }
1841 
1842     var allTasksCompleted = true;
1843 
1844     self.each(function(task) {
1845       if (!task.completed) {
1846         allTasksCompleted = false;
1847       }
1848     });
1849 
1850     if (allTasksCompleted) {
1851       self.tasksCompleted = true;
1852       Krypsis.fireEvent(self, 'onTasksCompletedEvent');
1853     }
1854   };
1855 
1856   this.each(function(task) {
1857     if (task instanceof Krypsis.Device.Task) {
1858       Krypsis.Log.debug("Array: execute task: " + task.toString());
1859       task.execute(self.taskCompleted);
1860     }
1861     else {
1862       Krypsis.Log.debug("Array: Object is not an instance of Krypsis.Device.Task: " + task.toString());
1863       throw new Error(task + " is not an instance of Krypsis.Device.Task");
1864     }
1865   });
1866 };

1869  * <p>
1870  * This function is usefull whenever you need some default values
1871  * in this object but on the other hand don't want to overwrite existing
1872  * values. 
1873  * </p>
1874  *
1875  * @example
1876  * <pre>
1877  * var myObj = { key: 'value' };
1878  * myObj.applyDefaults({ key: 'another value', anotherKey: 'third value'});
1879  *
1880  * // myObj => {key: 'value', anotherKey: 'third value'} 
1881  * </pre>
1882  *
1883  * @param {Object} defaultValues An object with default values to apply to this object
1884  * @public
1885  */
1886 Object.prototype.applyDefaults = function(defaultValues) {
1887   defaultValues = defaultValues || {};
1888   for (var key in defaultValues) {
1889     var value = this[key];
1890 
1891     if (typeof value == 'undefined') {
1892       this[key] = defaultValues[key];
1893     }
1894   }
1895   return this;
