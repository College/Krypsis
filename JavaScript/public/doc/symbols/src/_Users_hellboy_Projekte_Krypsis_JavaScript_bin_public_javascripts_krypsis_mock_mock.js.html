 26 if (!Krypsis.isDefined(Krypsis.Mock)) {

 28    * @class The mocking library of Krypsis.
 29    * @static

 31   Krypsis.Mock = {};
 32 }
 33 if (!Krypsis.isDefined(Krypsis.Mock.Device)) {
 34   Krypsis.Mock.Device = {};

 37 /**
 38  * <p>
 39  * The mock namespace defines objects to mock the entire Krypsis
 40  * JavaScript library. The main purpose is to let web developers
 41  * create theier web application without the need of a real device.
 42  * </p>
 43  *
 44  * <p>
 45  * You are free to extend this library to fit you need. At the present
 46  * the library mocks the successful case of an execution process. To mock
 47  * a module you just need to include the JavaScript file in the "mock/success"
 48  * folder. 
 49  * </p>
 50  *
 51  * @public
 52  * @static
 53  * @class The Krypsis JavaScript mocking library
 54  */
 55 Krypsis.Mock.Device = {
 56 
 57   /**
 58    * <p>
 59    * The connector mock is an implementation of the connector
 60    * interface and mocks up the execute function to provide
 61    * dummy data. Include this mock into your page when you
 62    * developing your krypsis application and play with that
 63    * fake data. It's much easier than to provide a real krypsis.
 64    * </p>
 65    *
 66    * @class Mock implementation of the connector interface
 67    * @public
 68    */
 69   Connector: {
 70     /**
 71      * <p>
 72      * The delegate is the original connector. It will be used to
 73      * pass the requested commands to if the module was not mocked.
 74      * </p>
 75      *
 76      * @private
 77      */
 78     delegate: null,
 79 
 80     /**
 81      * <p>
 82      * Execute the function and pass the call to the mocked function if
 83      * exists. If the module is not mocked the call will be delegated to the
 84      * original connector.
 85      * </p>
 86      *
 87      * @param context The calling module
 88      * @param command The command to execute
 89      * @param args The arguments to pass to the device.
 90      *
 91      * @public
 92      */
 93     execute: function(context, command, args) {
 94       /* Get the modulename and upcase the first letter */
 95       var modulename = context.substring(0, 1).toUpperCase() + context.substring(1);
 96 
 97       /*
 98        If this module is not mocked, then delegate the command to the
 99        original connector. If the connector does not exist, then throw an exception
100        */
101       var mock = Krypsis.Mock.Device[modulename];
102       if (!mock) {
103         if (this.delegate) {
104           this.delegate.execute(context, command, args);
105         }
106         else {
107           throw new Error("Context '" + context + "' is not mocked and there is no connector to delegate to.");
108         }
109       }
110 
111 
112       if (mock[command]) {
113         mock[command](args);
114       }
115       else {
116         throw new Error("Mock does exist but does not provide the method: " + command);
117       }
118     }
119   }
120 };
121 
122 /* Save the original connector */
123 Krypsis.Mock.Device.Connector.delegate = Krypsis.Device.guessConnector() || null;
124 
125 /* Mock the device connector */
126 Krypsis.Device.connector = Krypsis.Mock.Device.Connector;
