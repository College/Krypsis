 23 /**
 24  * <p>
 25  * The default implementation of Krypsis logging interface does
 26  * nothing. It means the developer has no possibility to get notified
 27  * when the framework is telling him something. 
 28  * </p>
 29  *
 30  * <p>
 31  * To change this behaviour you could implement the Adapter interface
 32  * and provide a log(level, message) method in your adapter implementation.
 33  * </p>
 34  *
 35  * @class The mocked Krypsis logger. 
 36  */
 37 Krypsis.Mock.Log = {
 38 
 39   /**
 40    * <p>
 41    * This adapter implements the Krypsis.Log.Adpater interface.
 42    * By using this you will get the output either in an element
 43    * you provide or the default element that will be created and
 44    * placed at the bottom of the page.
 45    * </p>
 46    *
 47    * <p>
 48    * The ID of the element that this adapter creates for you is
 49    * <code>KrypsisMockLogger</code>
 50    * </p>
 51    */
 52   adapter: {
 53     /**
 54      * <p>
 55      * If set to true, the the adapter is initialized
 56      * and can be used. Default is FALSE
 57      * </p>
 58      *
 59      * @private
 60      */
 61     initialized: false,
 62 
 63     /**
 64      * <p>
 65      * The element to append the log message to.
 66      * The default is an element with the ID 'logging'
 67      * You can overwrite this property to provide your own
 68      * container.
 69      * </p>
 70      *
 71      * @example
 72      * <pre>
 73      *  Krypsis.Mock.Log.adapter.output = 'your_element_id';
 74      * </pre>
 75      * 
 76      * @public
 77      */
 78     output: 'KrypsisMockLogger',
 79 
 80     /**
 81      * <p>
 82      * Write the given message to the logger. The message will
 83      * be prefixed with the current date and the given log level.
 84      * </p>
 85      *
 86      * @param level The level of the message
 87      * @param args The message itself.
 88      *
 89      * @public
 90      */
 91     log: function(level, args) {
 92       /* Do nothing if the output is not specified. */
 93       if (!this.initialized) {
 94         this.initialize(); 
 95       }
 96 
 97       /*
 98       If output element does not exists,
 99       we do not have a target to write our message to
100       */
101       if (!this.output) {
102         return null;
103       }
104 
105       try {
106         var logName = Krypsis.Log.levelName(level);
107         var date = new Date();
108 
109         var message = "[" + this.getHours(date) + ":" + this.getMinutes(date) + ":" + this.getSeconds(date) + "] " + logName + ": " + args + "<br />" + this.output.innerHTML;
110         if (message.length > 1000) {
111           message = message.substring(0, 1000) + "...";
112         }
113         this.output.innerHTML = message;
114       }
115       catch(e) {
116         alert("The mock logging interface catched an exception: " + e.message);
117       }
118     },
119 
120     /**
121      * @private
122      */
123     getHours: function(date) {
124       var hours = date.getHours().toString();
125       return this.normalize(hours);
126     },
127 
128     /**
129      * @private
130      */
131     getMinutes: function(date) {
132       var minutes = date.getMinutes().toString();
133       return this.normalize(minutes);
134     },
135 
136     /**
137      * @private
138      */
139     getSeconds: function(date) {
140       var seconds = date.getSeconds().toString();
141       return this.normalize(seconds);
142     },
143 
144     /**
145      * @private
146      */
147     normalize: function(value) {
148       if (value.length == 1) {
149         value = "0" + value;
150       }
151       return value;
152     },
153 
154     /**
155      * <p>
156      * Initialize the adapter by requesting the DOM
157      * element that is specified by the 'output' field.
158      * If the element does not exists, then it will be created and placed
159      * just before the closing </body> Tag.
160      * </p>
161      *
162      * <p>
163      * You do not have to call this function manually. It will be done
164      * by the first log call.
165      * </p>
166      *
167      * @public
168      */
169     initialize: function() {
170       if (this.output) {
171         try {
172           this.output = Krypsis.$(this.output);
173         }
174         catch(e) {
175           var div = document.createElement("div");
176           div.id = this.output;
177           document.body.appendChild(div);
178           this.output = div;
179         }
180       }
181 
182       this.initialized = true;
183     }
184   } 
185 };
186 
187 /* Replace the default adapter by this mock adapter */
188 Krypsis.Log.adapter = Krypsis.Mock.Log.adapter;
189 Krypsis.Log.level = Krypsis.Log.DEBUG;
