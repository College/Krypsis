 23 /**
 24  * <p>
 25  * The phone module gives you access to the phone core
 26  * functionality like sending sms or dialing a phone number.
 27  * </p>
 28  *
 29  * @static
 30  * @class The phone module to access the phone core functions
 31  * @since 0.2
 32  * @depends ../krypsis.js
 33  */
 34 Krypsis.Device.Phone = {
 35   /**
 36    * @private
 37    */
 38   context: 'phone',
 39 
 40   /**
 41    * <p>
 42    * Sends a short message to a receiver.
 43    * </p>
 44    *
 45    * <p>
 46    * The expected parameters are the phone number to send the message
 47    * to and the text to send.
 48    *
 49    * <ul>
 50    * <li><strong>{String} number</strong> The phone number of the receiver. Required!</li>
 51    * <li><strong>{String} text</strong> The text to send. Required!</li>
 52    * </ul>
 53    * </p>
 54    *
 55    * @example
 56    * <pre>
 57    *  Krypsis.Device.Phone.sendSms({
 58    *    parameters: {
 59    *      number: '+4912234466887',
 60    *      text:   'Hi, I will be there soon ;)'
 61    *    },
 62    *    onSuccess: function() {
 63    *      alert("SMS sent successfully");
 64    *    }
 65    *  });
 66    *
 67    *  // or
 68    *
 69    *  var task = Krypsis.Device.Phone.sendSms({...}, false);
 70    *  task.execute();
 71    * </pre>
 72    *
 73    * @param {Object} options The options map with the parameters
 74    * and the callbacks.
 75    * @param {Boolean} execute Shold the task be executed after creation. Default is true
 76    *
 77    * @since 0.2
 78    * @public
 79    */
 80   sendSms: function(options, execute) {
 81     return Krypsis.Device.createTask(this, "sendsms", options, execute);
 82   },
 83 
 84   /**
 85    * <p>
 86    * Make a phone call to a particular number.
 87    * Note that the user can cancel this operation.
 88    * </p>
 89    *
 90    * <p>
 91    * <em>Parameters</em>
 92    * <ul>
 93    * <li><strong>{String} number</strong> The phone number. There are no limitations on the
 94    * number format, but make sure the number is valid. Otherwise an error will be thrown</li>
 95    * </ul>
 96    *
 97    * <em>Callbacks</em>
 98    * <ul>
 99    * <li><strong>onSuccess</strong> Will be called when
100    * the user accepts the number and before the dialing begins.</li>
101    * <li><strong>onFailure</strong> Is executed when the phone number is malformed or
102    * the user canceled the operation.</li>
103    * </ul>
104    * </p>
105    *
106    * @example
107    * <pre>
108    *  Krypsis.Device.Phone.dialNumber({
109    *    number: '00491234123456'
110    *  });
111    *
112    *  // or
113    *
114    *  var task = Krypsis.Device.Phone.dialNumber({...}, false);
115    *  task.execute();
116    * </pre>
117    *
118    *
119    * @param {Object} options The parameters and callbacks
120    * @param {Boolean} execute Shold the task be executed after creation. Default is true
121    *
122    * @since 0.2
123    * @public
124    */
125   dialNumber: function(options, execute) {
126     return Krypsis.Device.createTask(this, "dialnumber", options, execute);
127   }
128 };
